Easy API Integration

API Example


// Get Virtual Number
const getNumber = async () => {
    const response = await fetch('https://api.smsprovider.com/v1/number', {
        method: 'POST',
        headers: {
            'Authorization': 'Bearer YOUR_API_KEY',
            'Content-Type': 'application/json'
        },
        body: JSON.stringify({
            service: 'whatsapp',
            country: 'IN'
        })
    });
    return await response.json();
};

// Check for OTP
const checkOTP = async (numberId) => {
    const response = await fetch(
        `https://api.smsprovider.com/v1/otp/${numberId}`
    );
    return await response.json();
};
